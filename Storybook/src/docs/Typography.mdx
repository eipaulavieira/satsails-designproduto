export const TH = ({ children, color }) => (
  <th style={{
    padding: '8px 14px', textAlign: 'left', fontSize: 10, fontWeight: 600,

);

export const TD = ({ children, mono, muted, style }) => (
  <td style={{
    padding: '10px 14px', fontSize: mono ? 12 : 13,
    fontFamily: mono ? 'JetBrains Mono, monospace' : 'Inter, sans-serif',
    color: muted ? '#9ca3af' : '#374151',
    borderBottom: '1px solid #f3f4f6',
    verticalAlign: 'middle',
    ...style,
  }}>{children}</td>
);

export const Pill = ({ children, color, bg, border }) => (
  <span style={{
    display: 'inline-block', padding: '2px 8px', borderRadius: 20,
    fontSize: 11, fontFamily: 'JetBrains Mono, monospace',
    fontWeight: 500, whiteSpace: 'nowrap',
    color: color || '#374151',
    background: bg || '#f3f4f6',
    border: `1px solid ${border || '#e5e7eb'}`,
  }}>{children}</span>
);

export const UsageRow = ({ px, freq, tier, where }) => {
  const rem = pxToRem(px);
  const theme = flutterTextTheme(px);
  const tierColor = tier === 1 ? '#F7931A' : tier === 2 ? '#FF9800' : tier === 3 ? '#D1D2D2' : '#727272';
  return (
    <tr>
      <TD mono style={{ fontWeight: 600, color: '#0C0B0D', fontSize: 13 }}>{px}px</TD>
      <TD mono><Pill>{rem}</Pill></TD>
      <TD mono><Pill color="#1565C0" bg="#E3F2FD" border="#90CAF9">{px} sp</Pill></TD>
      <TD mono muted>{theme || '—'}</TD>
      <TD><span style={{ fontSize: 11, fontFamily: 'JetBrains Mono, monospace', color: tierColor, fontWeight: 600 }}>{freq}</span></TD>
      <TD muted style={{ fontSize: 12 }}>{where}</TD>
    </tr>
  );
};
